<script>
function filterTable() {
    const query = document.getElementById("stationSearch").value.toLowerCase().trim();
    const rows  = document.querySelectorAll("#tableBody tr");
    let visible = 0;
    rows.forEach(row => {
        const match = row.cells[1].textContent.toLowerCase().includes(query);
        row.style.display = match ? "" : "none";
        if (match) visible++;
    });
    document.getElementById("noMatch").style.display = visible === 0 ? "block" : "none";
}
</script>
